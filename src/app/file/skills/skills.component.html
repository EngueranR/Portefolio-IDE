public class Skill {


      private String language; // langage de programmation

      private String sourceIcon; // source pour icon


      public Skills(String language, String sourceIcon) {

      this.language = language;

      this.sourceIcon = sourceIcon;

      }


      public String toString() {

      return "language : " + language + '\n' +

      "sourceIcon : " + sourceIcon + "\\n\\n";

      }


      }
