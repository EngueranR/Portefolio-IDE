public class Experience {

      private String year; // Année

      private String role; // Rôle occupé

      private String company; // Entreprise

      private String description; // Description du poste

      public experiences ( String year, String role, String company, String description) {

      this.year = year;

      this.role = role;

      this.company = company;

      this.description = description;

      }

      public String toString() {

      return "year : " + year + '\n' +

      "role : " + role + '\n' +

      "company : " + company + '\n' +

      "description : " + description + "\\n\\n";

      }

      }
