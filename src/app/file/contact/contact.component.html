public class Contact {

      private String type; // type de contact

      private String value; // valeur du type

      private String sourceIcon; // source pour icon


      public Contact(String type, String value, String sourceIcon) {

      this.type = type;

      this.value = value;

      this.sourceIcon = sourceIcon;

      }


      public String toString() {

      return "type : " + type + '\n' +

      "value : " + value + '\n' +

      "sourceIcon : " + sourceIcon + "\\n\\n";

      }


      }
