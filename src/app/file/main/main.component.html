import com.fasterxml.jackson.databind.JsonNode;

      import com.fasterxml.jackson.databind.ObjectMapper;

      import java.io.IOException;

      import java.nio.file.Paths;

      import java.util.ArrayList;

      import java.util.List;


      public class Main {

      public static void main(String[] args) {

      ObjectMapper mapper = new ObjectMapper();

      List<Experiences> experiences = new ArrayList<>();

      List<Cursus> cursusList = new ArrayList<>();

      List<Skills> skillsList = new ArrayList<>();

      List<Contact> contactList = new ArrayList<>();


      try {

      // Lire le fichier JSON

      JsonNode rootNode = mapper.readTree(Paths.get("src/assets/json/experiences.json").toFile());

      // Parcourir le tableau "experience" et créer les objets Experiences

      if (rootNode.has("experience")) {

      JsonNode experienceNode = rootNode.get("experience");

      for (JsonNode node : experienceNode) {

      String year = node.get("year").asText();

      String role = node.get("role").asText();

      String company = node.get("company").asText();

      String description;


      if (node.get("description").isArray()) {

      List<String> descriptions = new ArrayList<>();

      for (JsonNode descNode : node.get("description")) {

      descriptions.add(descNode.asText());

      }

      description = String.join(", ", descriptions);

      } else {

      description = node.get("description").asText();

      }


      // Créer un nouvel objet Experiences et l'ajouter à la liste

      experiences.add(new Experiences(year, role, company, description));

      }

      }


      // Lire le deuxième fichier JSON

      rootNode = mapper.readTree(Paths.get("src/assets/json/cursus.json").toFile());


      // Parcourir le tableau "education" et créer les objets Cursus

      if (rootNode.has("education")) {

      JsonNode educationNode = rootNode.get("education");

      for (JsonNode node : educationNode) {

      String year = node.get("year").asText();

      String description = node.get("description").asText();

      String institution = node.get("institution").asText();


      // Créer un nouvel objet Cursus et l'ajouter à la liste

      cursusList.add(new Cursus(year, description, institution));

      }

      }


      // Lire le troisième fichier JSON

      rootNode = mapper.readTree(Paths.get("src/assets/json/skills.json").toFile());


      // Parcourir le tableau "skill" et créer les objets Skills

      if (rootNode.has("skill")) {

      JsonNode skillNode = rootNode.get("skill");

      for (JsonNode node : skillNode) {

      String language = node.get("language").asText();

      String sourceIcon = node.get("sourceIcon").asText();


      // Créer un nouvel objet Skills et l'ajouter à la liste

      skillsList.add(new Skills(language, sourceIcon));

      }

      }


      // Lire le quatrième fichier JSON

      rootNode = mapper.readTree(Paths.get("src/assets/json/contact.json").toFile());


      // Parcourir le tableau "details" et créer les objets Contact

      if (rootNode.has("details")) {

      JsonNode detailsNode = rootNode.get("details");

      for (JsonNode node : detailsNode) {

      String type = node.get("type").asText();

      String value = node.get("value").asText();

      String sourceIcon = node.get("sourceIcon").asText();


      // Créer un nouvel objet Contact et l'ajouter à la liste

      contactList.add(new Contact(type, value, sourceIcon));

      }

      }


      // Afficher les objets Experiences

      for (Experiences e : experiences) {

      System.out.println(e);

      }


      // Afficher les objets Cursus

      for (Cursus c : cursusList) {

      System.out.println(c);

      }


      // Afficher les objets Skills

      for (Skills s : skillsList) {

      System.out.println(s);

      }


      // Afficher les objets Contact

      for (Contact c : contactList) {

      System.out.println(c);

      }


      } catch (IOException e) {

      e.printStackTrace();

      }

    }
}
